7. Write a C program to enter length in centimeter and convert it into meter and
kilometer.

#include<stdio.h>
int main(){
    printf("Enter the number");
    float cm;
    scanf("%f", &cm);
    
    float m = cm/100;
    printf("%f\n", m);
    
    float km = m/1000;
    printf("%f", km);
    
    return 0;
}
